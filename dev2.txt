(async function () {

    // ─── CONFIGURATION ────────────────────────────────────────────────────────
    const WAIT_FOR_LINK   = 5000;   // ms to wait after clicking before checking
    const RETRY_WAIT      = 4000;   // ms to wait on each retry
    const MAX_RETRIES     = 3;      // how many extra attempts before giving up
    const PAGE_WAIT       = 7000;   // ms to wait after clicking Next Page
    // ──────────────────────────────────────────────────────────────────────────

    let fullLog   = [];
    let pageCount = 1;

    // ── STEP 1: PROPER PER-CLICK PROMISE-BASED INTERCEPTOR ────────────
    const waiters = []; // array of { resolve, reject, timeoutId }

    function notifyWaiters(url) {
        if (!url.includes("index.m3u8")) return;
        if (waiters.length > 0) {
            const waiter = waiters.shift();
            clearTimeout(waiter.timeoutId);
            waiter.resolve(url);
        }
    }

    const originalOpen = XMLHttpRequest.prototype.open;
    XMLHttpRequest.prototype.open = function (method, url) {
        if (typeof url === "string") notifyWaiters(url);
        return originalOpen.apply(this, arguments);
    };

    const originalFetch = window.fetch;
    window.fetch = async function (...args) {
        const url = args[0] ? args[0].toString() : "";
        notifyWaiters(url);
        return originalFetch.apply(this, args);
    };

    function waitForLink(ms) {
        return new Promise((resolve) => {
            const timeoutId = setTimeout(() => {
                const idx = waiters.findIndex(w => w.resolve === resolve);
                if (idx !== -1) waiters.splice(idx, 1);
                resolve(null); 
            }, ms);
            waiters.push({ resolve, reject: null, timeoutId });
        });
    }

    // ── STEP 2: HELPERS ───────────────────────────────────────────────────────

    function checkForYouTube() {
        const iframes = document.querySelectorAll("iframe");
        for (let iframe of iframes) {
            if (iframe.src.includes("youtube") || iframe.src.includes("youtu.be")) {
                return true;
            }
        }
        return false;
    }

    function downloadTxtFile() {
        const blob = new Blob([fullLog.join("\n\n")], { type: "text/plain" });
        const a    = document.createElement("a");
        a.href     = URL.createObjectURL(blob);
        a.download = `Course_Links_${Date.now()}.txt`;
        document.body.appendChild(a);
        a.click();
        document.body.removeChild(a);
    }

    const sleep = ms => new Promise(r => setTimeout(r, ms));

    // Fix 1: Auto-scrolling to trigger lazy-loaded elements
    async function scrollToBottom() {
        let lastHeight = document.body.scrollHeight;
        while (true) {
            window.scrollTo(0, document.body.scrollHeight);
            await sleep(1500); // Wait for lazy load to trigger
            let newHeight = document.body.scrollHeight;
            if (newHeight === lastHeight) break; // Reached bottom
            lastHeight = newHeight;
        }
        // Scroll back to top to ensure clicks register properly
        window.scrollTo(0, 0);
        await sleep(500);
    }

    // ── STEP 3: CLICK + WAIT + RETRY FOR A SINGLE VIDEO ──────────────────────

    async function processVideo(container, index) {
        const titleEl = container.querySelector("p.break-all");
        const title   = titleEl ? titleEl.innerText.trim() : `Video ${index + 1}`;

        let finalLink  = null;
        let isYouTube  = false;

        for (let attempt = 0; attempt <= MAX_RETRIES; attempt++) {
            if (attempt > 0) {
                console.log(`%c  ↻ Retry ${attempt}/${MAX_RETRIES} for: "${title}"`, "color: orange");
                await sleep(500); 
            }

            const clickTarget = container.querySelector("div.flex.justify-between") || container;
            
            // Bring element into view before clicking (helps with stubborn UI frameworks)
            clickTarget.scrollIntoView({ behavior: "smooth", block: "center" });
            await sleep(300);
            
            clickTarget.click();

            const waitMs = attempt === 0 ? WAIT_FOR_LINK : RETRY_WAIT;
            finalLink    = await waitForLink(waitMs);

            if (finalLink) break; 

            if (checkForYouTube()) {
                isYouTube = true;
                break;
            }
        }

        let displayLink, statusColor;
        if (finalLink) {
            displayLink = finalLink;
            statusColor = "color: #00ff00";
        } else if (isYouTube) {
            displayLink = "YouTube Video";
            statusColor = "color: orange";
        } else {
            displayLink = "Link not found (exhausted retries)";
            statusColor = "color: red";
        }

        console.log(`%c${title}`, "font-weight: bold; color: white;");
        console.log(`%c${displayLink}`, statusColor);
        console.log("───────────────────────────────────");

        fullLog.push(`${title}\n${displayLink}`);
    }

    // ── STEP 4: MAIN LOOP ─────────────────────────────────────────────────────

    console.clear();
    console.log(
        "%cSTARTING SCRAPER (bulletproof) — DO NOT CLOSE THIS TAB.",
        "color: #00ffff; font-size: 16px; font-weight: bold;"
    );

    let keepGoing = true;

    while (keepGoing) {
        
        console.log(`%cScrolling to load all items on Page ${pageCount}...`, "color: gray");
        await scrollToBottom(); // Ensures all videos on this page are in the DOM

        let videoListItems = document.querySelectorAll(".video-list");
        console.log(
            `%c─── PAGE ${pageCount} (${videoListItems.length} videos) ───`,
            "color: yellow; font-weight: bold;"
        );

        for (let i = 0; i < videoListItems.length; i++) {
            // Fix 2: Re-query the DOM inside the loop to avoid Stale/Detached Element errors
            let freshDomItems = document.querySelectorAll(".video-list");
            if (freshDomItems[i]) {
                await processVideo(freshDomItems[i], i);
            } else {
                console.log(`%cElement ${i} vanished from DOM.`, "color: red");
            }
        }

        // ── Pagination ──
        const nextLi   = document.querySelector("li.next");
        const nextBtn  = nextLi ? nextLi.querySelector("a") : null;
        const isLastPage =
            !nextBtn ||
            nextLi.classList.contains("disabled")  ||
            nextLi.classList.contains("invisible") ||
            nextLi.getAttribute("aria-disabled") === "true";

        if (!isLastPage) {
            console.log(`%cPage ${pageCount} done → loading page ${pageCount + 1}…`, "color: cyan");

            const oldFirstTitle = document.querySelector(".video-list p.break-all")?.innerText;
            
            // Ensure next button is in view before clicking
            if(nextBtn.scrollIntoView) nextBtn.scrollIntoView({ block: "center" });
            await sleep(500);
            
            nextBtn.click();
            await sleep(PAGE_WAIT);

            const newFirstTitle = document.querySelector(".video-list p.break-all")?.innerText;
            if (oldFirstTitle === newFirstTitle) {
                console.log("%cPage content unchanged — assuming last page.", "color: red");
                keepGoing = false;
            } else {
                pageCount++;
            }
        } else {
            console.log("%cNo Next button — end of course.", "color: cyan");
            keepGoing = false;
        }
    }

    // ── Finish ──
    console.log("%c✅ DONE! Generating file…", "color: #00ff00; font-size: 18px;");
    downloadTxtFile();

})();
